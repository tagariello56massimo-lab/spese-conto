<script>
var spese = JSON.parse(localStorage.getItem('spese_condo_v3') || '[]');
var tabAttivo = 'tutte';
var filtroAttivo = 'tutte';
var pagataToggle = false;
var fileAllegato = null;
var editId = null;

function salva() { localStorage.setItem('spese_condo_v3', JSON.stringify(spese)); }
function oggi() { return new Date().toISOString().split('T')[0]; }

function isScaduta(s) { return !s.pagata && s.scadenza && s.scadenza < oggi(); }
function isInScadenza(s) {
  if (s.pagata || !s.scadenza) return false;
  var diff = (new Date(s.scadenza) - new Date()) / (1000*60*60*24);
  return diff >= 0 && diff <= 7;
}

function fmt(n) {
  return '€' + Number(n).toLocaleString('it-IT', {minimumFractionDigits:2, maximumFractionDigits:2});
}
function fmtDate(d) {
  if (!d) return '';
  var p = d.split('-');
  return p[2]+'/'+p[1]+'/'+p[0];
}
function esc(str) {
  return String(str).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}

function switchTab(tab) {
  tabAttivo = tab;
  document.getElementById('tab-tutte').classList.toggle('active', tab === 'tutte');
  document.getElementById('tab-casa1').classList.toggle('active', tab === 'casa1');
  document.getElementById('tab-montagna').classList.toggle('active', tab === 'montagna');
  renderLista();
}

function setFilter(f) {
  filtroAttivo = f;
  ['tutte','pagate','nonpagate','scadute'].forEach(function(x) {
    document.getElementById('f-'+x).classList.toggle('active', x === f);
  });
  renderLista();
}

function renderLista() {
  var lista = spese.slice();

  // Filtra per tab immobile
  if (tabAttivo === 'casa1') {
    lista = lista.filter(function(s) { return s.immobile === 'casa1'; });
  } else if (tabAttivo === 'montagna') {
    lista = lista.filter(function(s) { return s.immobile === 'montagna'; });
  }

  // Filtra per stato pagamento
  if (filtroAttivo === 'pagate') {
    lista = lista.filter(function(s) { return s.pagata; });
  } else if (filtroAttivo === 'nonpagate') {
    lista = lista.filter(function(s) { return !s.pagata; });
  } else if (filtroAttivo === 'scadute') {
    lista = lista.filter(function(s) { return isScaduta(s); });
  }

  lista.sort(function(a,b) {
    var da = a.scadenza || '9999';
    var db = b.scadenza || '9999';
    return da < db ? -1 : da > db ? 1 : 0;
  });

  // Calcola totali sul tab corrente (non su filtro)
  var listaTab = tabAttivo === 'tutte' ? spese : spese.filter(function(s){ return s.immobile === tabAttivo; });
  var totale = 0, pagato = 0, nonpagato = 0, inScad = 0;
  listaTab.forEach(function(s) {
    var imp = parseFloat(s.importo) || 0;
    totale += imp;
    if (s.pagata) pagato += imp; else nonpagato += imp;
    if (isScaduta(s) || isInScadenza(s)) inScad++;
  });
  document.getElementById('tot-totale').textContent = fmt(totale);
  document.getElementById('tot-pagato').textContent = fmt(pagato);
  document.getElementById('tot-nonpagato').textContent = fmt(nonpagato);
  document.getElementById('tot-scadenza').textContent = inScad;

  var el = document.getElementById('lista-spese');
  if (!lista.length) {
    el.innerHTML = '<div class="empty"><div class="empty-icon">📋</div><p>Nessuna spesa trovata</p></div>';
    return;
  }

  el.innerHTML = lista.map(function(s) {
    var scad = isScaduta(s);
    var vicina = isInScadenza(s);
    return '<div class="spesa-card' + (scad ? ' scaduta' : '') + '">'
      + '<div class="spesa-top">'
      + '<div class="spesa-nome">' + esc(s.nome) + '</div>'
      + '<div class="spesa-importo">' + fmt(s.importo) + '</div>'
      + '</div>'
      + '<div class="spesa-meta">'
      + '<span class="badge ' + s.immobile + '">' + (s.immobile === 'casa1' ? 'Prima casa' : 'Montagna') + '</span>'
      + '<span class="badge ' + (s.pagata ? 'pagata' : 'nonpagata') + '">' + (s.pagata ? 'Pagata' : 'Da pagare') + '</span>'
      + (s.scadenza ? '<span class="spesa-date">Scad. ' + fmtDate(s.scadenza) + '</span>' : '')
      + '</div>'
      + (s.iban ? '<div class="spesa-iban">IBAN <span>' + esc(s.iban) + '</span></div>' : '')
      + (scad ? '<div class="scadenza-warning">⚠ Scadenza superata</div>' : '')
      + (vicina ? '<div class="scadenza-warning">⚠ Scade entro 7 giorni</div>' : '')
      + (s.fileName ? '<div class="spesa-attach">📎 ' + esc(s.fileName) + '</div>' : '')
      + '<div class="spesa-actions">'
      + (!s.pagata
          ? '<button class="btn-small success" onclick="marcaPagata(\'' + s.id + '\')">Segna pagata</button>'
          : '<button class="btn-small" onclick="marcaNonPagata(\'' + s.id + '\')">Non pagata</button>')
      + '<button class="btn-small" onclick="editSpesa(\'' + s.id + '\')">Modifica</button>'
      + '<button class="btn-small danger" onclick="eliminaSpesa(\'' + s.id + '\')">Elimina</button>'
      + '</div></div>';
  }).join('');
}

function marcaPagata(id) { var s = spese.find(function(x){return x.id===id;}); if(s){s.pagata=true; salva(); renderLista();} }
function marcaNonPagata(id) { var s = spese.find(function(x){return x.id===id;}); if(s){s.pagata=false; salva(); renderLista();} }
function eliminaSpesa(id) { if(confirm('Eliminare questa spesa?')){spese=spese.filter(function(x){return x.id!==id;}); salva(); renderLista();} }
function editSpesa(id) { openModal(id); }

function openModal(id) {
  editId = id || null;
  pagataToggle = false;
  fileAllegato = null;
  document.getElementById('attach-name').textContent = '';
  document.getElementById('iban-hint').textContent = 'Facoltativo';
  document.getElementById('iban-hint').className = 'hint';
  document.getElementById('modal-title').textContent = id ? 'Modifica spesa' : 'Nuova spesa';
  if (id) {
    var s = spese.find(function(x){return x.id===id;});
    document.getElementById('f-nome').value = s.nome;
    document.getElementById('f-importo').value = s.importo;
    document.getElementById('f-iban').value = s.iban || '';
    document.getElementById('f-immobile').value = s.immobile;
    document.getElementById('f-scadenza').value = s.scadenza || '';
    pagataToggle = s.pagata;
    if (s.fileName) document.getElementById('attach-name').textContent = '📎 ' + s.fileName;
  } else {
    document.getElementById('f-nome').value = '';
    document.getElementById('f-importo').value = '';
    document.getElementById('f-iban').value = '';
    document.getElementById('f-immobile').value = tabAttivo !== 'tutte' ? tabAttivo : 'casa1';
    document.getElementById('f-scadenza').value = '';
  }
  document.getElementById('toggle-pagata').className = 'toggle' + (pagataToggle ? ' on' : '');
  document.getElementById('modal-overlay').classList.add('open');
}

function closeModal() { document.getElementById('modal-overlay').classList.remove('open'); editId = null; }
function overlayClick(e) { if (e.target === document.getElementById('modal-overlay')) closeModal(); }

function togglePagata() {
  pagataToggle = !pagataToggle;
  document.getElementById('toggle-pagata').className = 'toggle' + (pagataToggle ? ' on' : '');
}

function handleFile(input) {
  if (input.files[0]) {
    fileAllegato = { name: input.files[0].name };
    document.getElementById('attach-name').textContent = '📎 ' + input.files[0].name;
  }
}

function formatIban(input) {
  var val = input.value.replace(/\s/g,'').toUpperCase().replace(/[^A-Z0-9]/g,'');
  var groups = val.match(/.{1,4}/g);
  input.value = groups ? groups.join(' ') : val;
  var hint = document.getElementById('iban-hint');
  if (!val) { hint.textContent = 'Facoltativo'; hint.className = 'hint'; return; }
  if (val.length < 15) { hint.textContent = 'IBAN troppo corto'; hint.className = 'hint invalid'; return; }
  var rear = val.slice(4) + val.slice(0,4);
  var num = rear.split('').map(function(c){ return isNaN(c) ? (c.charCodeAt(0)-55).toString() : c; }).join('');
  var rem = 0;
  for (var i = 0; i < num.length; i++) rem = (rem * 10 + parseInt(num[i])) % 97;
  if (rem === 1) { hint.textContent = 'IBAN valido ✓'; hint.className = 'hint valid'; }
  else { hint.textContent = 'IBAN non valido — controlla i dati'; hint.className = 'hint invalid'; }
}

function saveSpesa() {
  var nome = document.getElementById('f-nome').value.trim();
  var importo = document.getElementById('f-importo').value;
  var iban = document.getElementById('f-iban').value.trim();
  var immobile = document.getElementById('f-immobile').value;
  var scadenza = document.getElementById('f-scadenza').value;
  if (!nome || !importo) { alert('Inserisci almeno descrizione e importo'); return; }
  if (editId) {
    var s = spese.find(function(x){return x.id===editId;});
    s.nome = nome; s.importo = parseFloat(importo); s.iban = iban;
    s.immobile = immobile; s.scadenza = scadenza; s.pagata = pagataToggle;
    if (fileAllegato) s.fileName = fileAllegato.name;
  } else {
    spese.push({
      id: Date.now().toString(), nome: nome,
      importo: parseFloat(importo), iban: iban,
      immobile: immobile, scadenza: scadenza,
      pagata: pagataToggle,
      fileName: fileAllegato ? fileAllegato.name : null,
      creata: oggi()
    });
  }
  salva(); closeModal(); renderLista();
}

function oggi() { return new Date().toISOString().split('T')[0]; }

renderLista();
</script>
